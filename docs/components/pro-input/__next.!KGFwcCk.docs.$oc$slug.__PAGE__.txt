## Usage

```tsx
import { Input } from "@/components/pro/pro-fields/input"
import { Password } from "@/components/pro/pro-fields/password"

export default function Demo() {
  return (
    <div className="flex flex-col gap-4">
      <Input placeholder="Basic input" />
      <Input prefix={<SearchIcon />} placeholder="With prefix icon" allowClear />
      <Password placeholder="Password" />
    </div>
  )
}
```
