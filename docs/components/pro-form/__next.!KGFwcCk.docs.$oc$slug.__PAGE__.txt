## Usage

```tsx
import { createForm, ProForm, SchemaField } from "@/components/ui/pro-form"

const form = createForm()

export default function BasicForm() {
  return (
    <ProForm form={form} onFinish={(values) => console.log(values)}>
      <SchemaField>
        <SchemaField.String
          name="username"
          title="Username"
          required
          x-decorator="FormItem"
          x-component="Input"
          x-component-props={{ placeholder: "Enter username" }}
        />
        <SchemaField.String
          name="email"
          title="Email"
          required
          x-decorator="FormItem"
          x-component="Input"
          x-component-props={{ type: "email", placeholder: "user@example.com" }}
        />
      </SchemaField>
    </ProForm>
  )
}
```

### Schema-Driven

Define your entire form with a JSON schema:

```tsx
import { createForm, ProForm, SchemaField } from "@/components/ui/pro-form"

const schema = {
  type: "object",
  properties: {
    username: {
      type: "string",
      title: "Username",
      required: true,
      "x-component": "Input",
      "x-validator": [{ required: true, message: "Username is required" }],
    },
    role: {
      type: "string",
      title: "Role",
      "x-component": "Select",
      enum: [
        { label: "Admin", value: "admin" },
        { label: "User", value: "user" },
        { label: "Guest", value: "guest" },
      ],
    },
    notifications: {
      type: "boolean",
      title: "Enable notifications",
      "x-component": "Switch",
    },
  },
}

const form = createForm()

export default function SchemaForm() {
  return (
    <ProForm form={form} onFinish={(values) => console.log(values)}>
      <SchemaField schema={schema} />
    </ProForm>
  )
}
```

### Field Description

Add a `description` property to a schema field to display helper text below the input:

```tsx
<SchemaField>
  <SchemaField.String
    name="email"
    title="Email"
    required
    description="We will never share your email with anyone."
    x-decorator="FormItem"
    x-component="Input"
    x-component-props={{ placeholder: 'user@example.com' }}
  />
  <SchemaField.String
    name="bio"
    title="Bio"
    description="Max 200 characters."
    x-decorator="FormItem"
    x-component="Textarea"
  />
  <SchemaField.Boolean
    name="active"
    title="Enable account"
    description="Disabled accounts cannot sign in."
    x-decorator="FormItem"
    x-component="Switch"
  />
</SchemaField>
```

## API Reference

### ProForm

| Prop | Type | Default | Description |
|---|---|---|---|
| `form` | `Form` | internal form | Formily form instance created with `createForm()`. Omit to let `ProForm` create one. |
| `formProps` | `IFormProps` | — | Options used when `ProForm` creates the internal form. |
| `children` | `ReactNode` | — | Form fields and layout components. |
| `onFinish` | `(values) => void \| Promise<void>` | — | Called after successful validation. |
| `onFinishFailed` | `(errors) => void` | — | Called when validation fails on submit. |
| `onReset` | `() => void \| Promise<void>` | — | Called after reset completes. |
| `submitText` | `string` | `'Submit'` | Label for the submit button. |
| `showReset` | `boolean` | `false` | Show a reset button alongside submit. |
| `resetText` | `string` | `'Reset'` | Label for the reset button. |
| `hideActions` | `boolean` | `false` | Hide the bottom action bar — place `<ProFormActions />` manually. |
| `actionsAlign` | `'left' \| 'center' \| 'right'` | `'left'` | Alignment of the action bar. |
| `columns` | `1 \| 2 \| 3 \| 4` | — | Multi-column layout. Wraps fields in `ProFormGrid` automatically. |
| `className` | `string` | — | Additional className for the `<form>` element. |

### ProFormGrid

Multi-column grid container — fields inside are evenly distributed.

| Prop | Type | Default | Description |
|---|---|---|---|
| `columns` | `1 \| 2 \| 3 \| 4` | `1` | Number of columns. |
| `gap` | `string` | `'gap-4'` | Tailwind gap class. |
| `className` | `string` | — | Additional className. |

### ProFormActions

Standalone action bar — use with `hideActions` to position it freely.

| Prop | Type | Default | Description |
|---|---|---|---|
| `submitText` | `string` | `'Submit'` | Label for the submit button. |
| `showReset` | `boolean` | `false` | Show a reset button. |
| `resetText` | `string` | `'Reset'` | Label for the reset button. |
| `onReset` | `() => void \| Promise<void>` | — | Reset callback. |
| `loading` | `boolean` | `false` | Shows loading state on the submit button. |
| `align` | `'left' \| 'center' \| 'right'` | `'left'` | Alignment of buttons. |
| `children` | `ReactNode` | — | Extra buttons or content to render alongside actions. |

export interface ProFormProps {
  form?: Form
  formProps?: IFormProps
  children?: ReactNode
  onFinish?: (values: Record<string, unknown>) => void | Promise<void>
  onFinishFailed?: (errors: unknown) => void
  onReset?: () => void | Promise<void>
  /** Submit button label (only effective when hideActions is false) */
  submitText?: string
  /** Whether to show the reset button */
  showReset?: boolean
  /** Reset button label */
  resetText?: string
  /** Hide the default action bar (place ProFormActions yourself) */
  hideActions?: boolean
  /** Action bar alignment */
  actionsAlign?: 'left' | 'center' | 'right'
  /** Number of form columns (passed through to ProFormGrid) */
  columns?: 1 | 2 | 3 | 4
  className?: string
}

export function ProForm({
  form,
  formProps,
  children,
  onFinish,
  onFinishFailed,
  onReset,
  submitText = 'Submit',
  showReset = false,
  resetText = 'Reset',
  hideActions = false,
  actionsAlign = 'left',
  columns,
  className,
}: ProFormProps) {
  const [loading, setLoading] = React.useState(false)
  const internalFormRef = React.useRef<Form | null>(null)

  if (!internalFormRef.current) {
    internalFormRef.current = createForm(formProps)
  }

  const activeForm = form ?? internalFormRef.current

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault()
    if (loading) return

    setLoading(true)
    try {
      await activeForm.validate()
      await onFinish?.(activeForm.values)
    } catch (err) {
      onFinishFailed?.(err)
    } finally {
      setLoading(false)
    }
  }

  async function handleReset() {
    await activeForm.reset()
    await onReset?.()
  }

  const body = columns ? (
    <ProFormGrid columns={columns} className="mb-4">
      {children}
    </ProFormGrid>
  ) : (
    <div className="space-y-4">{children}</div>
  )

  return (
    <FormProvider form={activeForm}>
      <form onSubmit={handleSubmit} className={className}>
        {body}
        {!hideActions && (
          <ProFormActions
            submitText={submitText}
            showReset={showReset}
            resetText={resetText}
            onReset={handleReset}
            loading={loading}
            align={actionsAlign}
          />
        )}
      </form>
    </FormProvider>
  )
}
