## Usage

```tsx
import { ProDescriptions } from "@/components/pro/pro-descriptions"

const items = [
  { label: "Name", value: "Alice Wang" },
  { label: "Email", value: "alice@example.com" },
  { label: "Role", value: "Admin" },
  { label: "Created At", value: new Date("2026-05-31").toLocaleDateString() },
]

export default function UserDetail() {
  return <ProDescriptions items={items} title="User Info" bordered />
}
```

## Examples

### With Custom Render

```tsx
const items = [
  { label: "Status", value: (
    <Badge variant="default">Active</Badge>
  )},
  { label: "Tags", value: (
    <div className="flex gap-1">
      {tags.map((tag) => <Badge key={tag} variant="outline">{tag}</Badge>)}
    </div>
  )},
]
```

### Multi-Column Layout

```tsx
<ProDescriptions items={items} columns={3} />
```

## API Reference

### ProDescriptions

| Prop | Type | Default | Description |
|---|---|---|---|
| `items` | `DescriptionsItem[]` | — | Description items to render. |
| `title` | `ReactNode` | — | Section title. |
| `columns` | `1 \| 2 \| 3 \| 4` | `2` | Number of columns per row. |
| `bordered` | `boolean` | `false` | Show bordered style |
| `layout` | `"horizontal" \| "vertical"` | `"horizontal"` | Label/value layout. |
| `className` | `string` | — | Root class name. |
| `labelClassName` | `string` | — | Class name for every label cell. |
| `valueClassName` | `string` | — | Class name for every value cell. |

### DescriptionsItem

| Prop | Type | Description |
|---|---|---|
| `label` | `ReactNode` | Item label. |
| `value` | `ReactNode` | Item value. Empty values render a muted dash. |
| `span` | `number` | Column span (default: 1) |
| `className` | `string` | Class name for this item. |
