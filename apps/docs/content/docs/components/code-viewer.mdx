## Usage

```tsx
import { CodeViewer } from "@/components/pro/pro-viewer/code"

export default function Example() {
  return (
    <CodeViewer
      code={`const hello = "world"`}
      lang="typescript"
      theme="dark"
      showLineNumbers
    />
  )
}
```

## API Reference

### CodeViewer

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `code` | `string` | — | The source code string to display. |
| `lang` | `string` | `"typescript"` | Language identifier for syntax highlighting (e.g. `"tsx"`, `"python"`). |
| `theme` | `"light" \| "dark"` | `"dark"` | Shared One syntax theme. |
| `showLineNumbers` | `boolean` | `true` | Whether to display line numbers. |
| `showHeader` | `boolean` | `true` | Whether to show the filename/language and copy toolbar. |
| `surface` | `"code" \| "embedded"` | `"code"` | Use `"embedded"` when rendering inside another surface. |
| `title` | `string` | — | Optional filename shown in the toolbar. |
| `emptyText` | `string` | `"No code"` | Message shown when `code` is empty. |
| `scrollRef` | `Ref<HTMLDivElement>` | — | Ref forwarded to the scrollable code viewport. |
| `onScroll` | `UIEventHandler<HTMLDivElement>` | — | Scroll handler for synchronizing with another viewer. |
| `className` | `string` | — | Additional CSS class for the root element. |
