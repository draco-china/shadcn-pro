## Installation

```bash
npx @draco-china/shadcn-pro add html-viewer
```

## Usage

```tsx
import { HtmlViewer } from '@/components/pro/pro-viewer/html'

export default function Example() {
  return (
    <HtmlViewer
      content="<h1 style='font-family:sans-serif'>Hello World</h1>"
      className="h-64 rounded-lg border"
    />
  )
}
```

## API Reference

### HtmlViewer

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `content` | `string` | — | Raw HTML string to render inside the sandboxed iframe. |
| `className` | `string` | `undefined` | Additional CSS classes applied to the iframe element. |
