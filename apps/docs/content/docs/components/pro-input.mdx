## Usage

```tsx
import { Input } from "@/components/pro/pro-fields/input"
import { Password } from "@/components/pro/pro-fields/password"

export default function Demo() {
  return (
    <div className="flex flex-col gap-4">
      <Input placeholder="Basic input" />
      <Input prefix={<SearchIcon />} placeholder="With prefix icon" allowClear />
      <Password placeholder="Password" />
    </div>
  )
}
```

## Props

| Prop | Type | Default | Description |
|---|---|---|---|
| `prefix` | `ReactNode` | — | Content to render before the input (icon or text). |
| `suffix` | `ReactNode` | — | Content to render after the input (icon or text). |
| `allowClear` | `boolean` | `false` | Show a clear button when the input has a value. |
| `onClear` | `() => void` | — | Called when the clear button is clicked. |
| `inputClassName` | `string` | — | CSS class passed to the inner shadcn Input when using `Password`. |
