## Usage

```tsx
import { DiffViewer } from "@/components/pro/pro-viewer/diff"

export default function Example() {
  return (
    <DiffViewer
      oldCode={`const a = 1`}
      newCode={`const a = 2`}
      lang="typescript"
      theme="github-dark"
      oldTitle="Before"
      newTitle="After"
    />
  )
}
```

## API Reference

### DiffViewer

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `oldCode` | `string` | — | The original (before) code. |
| `newCode` | `string` | — | The modified (after) code. |
| `lang` | `string` | `"typescript"` | Language for syntax highlighting. |
| `theme` | `"github-light" \| "github-dark"` | `"github-dark"` | Shared GitHub syntax theme used by Shiki. |
| `oldTitle` | `string` | `"Before"` | Label shown above the original code panel. |
| `newTitle` | `string` | `"After"` | Label shown above the modified code panel. |
| `defaultView` | `"split" \| "unified"` | `"split"` | Initial view mode. |
| `className` | `string` | — | Additional CSS class for the root element. |
