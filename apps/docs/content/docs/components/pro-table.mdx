## Usage

```tsx
import { ProTable } from "@/components/pro/pro-table"

const columns = [
  {
    accessorKey: "name",
    header: "Name",
    enableSorting: true,
    meta: { pinned: "left", search: { placeholder: "Search by name..." } },
  },
  { accessorKey: "email", header: "Email" },
  { accessorKey: "createdAt", header: "Created", meta: { pinned: "right" } },
]

export default function UsersTable() {
  return (
    <ProTable
      columns={columns}
      data={users}
      header={<h2 className="text-lg font-semibold">Users</h2>}
      stickyHeader
      toolbar={{ options: { refresh: () => refetch() } }}
    />
  )
}
```

## Examples

### Pinned Columns and Sticky Header

Use `meta.pinned` on column definitions to keep important columns visible during horizontal scroll. Users can also pin or unpin columns from the column settings menu. The table header is sticky by default; set `stickyHeader={false}` to disable it.

```tsx
const columns = [
  { id: "select", size: 48, meta: { pinned: "left" } },
  { accessorKey: "name", header: "Name", size: 180, meta: { pinned: "left" } },
  { accessorKey: "email", header: "Email", size: 240 },
  { accessorKey: "createdAt", header: "Created", size: 150, meta: { pinned: "right" } },
]

<ProTable columns={columns} data={users} />
```

### Filtering with Reset

Add `meta.search` or `meta.filters` to columns — ProTable **automatically** renders the toolbar search input and `FacetedFilter` controls, maps `value → label` in cells, and shows a **Reset** button whenever any filter is active.

- Default mode is **multi-select** — use `filterFn: "multiValueFilter"` (built-in)
- Set `meta.filterMode: "single"` for single-select (radio-like), use `filterFn: "equals"`

<ComponentPreview name="pro-table-filter-demo" previewClassName="min-h-[460px]" />

```tsx
const columns: ColumnDef<User>[] = [
  {
    accessorKey: "name",
    header: "Name",
    meta: {
      search: { placeholder: "Search name..." },
    },
  },
  {
    accessorKey: "role",
    header: "Role",
    filterFn: "multiValueFilter", // built-in, handles string[] filter value
    meta: {
      filterPlaceholder: "Role",
      filters: [
        { label: "Admin", value: "admin" },
        { label: "Editor", value: "editor" },
      ],
    },
  },
  {
    accessorKey: "status",
    header: "Status",
    filterFn: "equals",
    meta: {
      filterPlaceholder: "Status",
      filterMode: "single",   // single-select
      filterVariant: "text",  // cell renders as plain text (default: 'badge')
      filters: [
        { label: "Active", value: "active" },
        { label: "Inactive", value: "inactive" },
      ],
    },
  },
]

// Search and filters are auto-rendered from column meta
<ProTable columns={columns} data={data} />
```

`meta.search` writes to the same column filter state as the rest of the toolbar. If you do not want to put search config in column meta, pass `toolbar={{ search: { columnId: "name", placeholder: "Search name..." } }}` instead.

### Bulk Actions

Configure bottom-floating bulk actions when rows are selected. The floating toolbar supports keyboard navigation and `Escape` clears the selection.

```tsx
<ProTable
  columns={columns}
  data={users}
  bulkToolbar={{
    entityName: "user",
    actions: [
      {
        key: "export-selected",
        label: ({ selectedRows }) => `Export ${selectedRows.length}`,
        icon: <Download size={16} />,
        tooltip: "Export selected users",
        onClick: ({ table }) => table.resetRowSelection(),
      },
    ],
  }}
/>
```

### Drag-to-Sort

Enable row drag-to-sort with the `dragSort` object. Use `rowKey` to specify the unique identifier field and `onDragSortEnd` to receive the reordered data.

Drag sorting reorders the backing `data` array by the currently rendered rows. When the table is filtered or paginated, the dragged row moves relative to the visible target row in the original data array.

<ComponentPreview
  name="pro-table-drag-sort-demo"

/>

```tsx
<ProTable
  columns={columns}
  data={data}
  dragSort={{
    enabled: true,
    rowKey: "id",
    onDragSortEnd: (newData) => console.log(newData),
  }}
/>
```

## API Reference

### ProTable

| Prop                    | Type                                  | Default                      | Description                                                                                                    |
| ----------------------- | ------------------------------------- | ---------------------------- | -------------------------------------------------------------------------------------------------------------- |
| `columns`               | `ColumnDef<T>[]`                      | —                            | TanStack Table column definitions.                                                                             |
| `data`                  | `T[]`                                 | `[]`                         | Table data array for local/client data.                                                                        |
| `request`               | `(state) => Promise<{ data, total? }>`| —                            | Server request function. When provided, pagination, sorting, and filtering are sent through request params.     |
| `defaultState`          | `Partial<ProTableState>`              | —                            | Initial pagination, sorting, and column filters. Useful for URL-derived initial state.                          |
| `onStateChange`         | `(state) => void`                     | —                            | Called when pagination, sorting, or filters change. Useful for URL sync or manually loading data.               |
| `header`                | `ReactNode \| (context) => ReactNode` | —                            | Custom content rendered above the toolbar.                                                                     |
| `toolbar`               | `false \| ProTableToolbarOptions<T>` | enabled                      | Toolbar config. Controls search, actions, and built-in options in one object.                                  |
| `toolbar.search`        | `false \| string \| { columnId, placeholder? }` | from `columns[].meta.search` | Explicit toolbar search config. Writes to the target column filter.                                  |
| `toolbar.options`       | `false \| { refresh?, density?, columns? }` | `{ density: true, columns: true }` | Built-in toolbar options, similar to Ant Design ProTable.                                           |
| `toolbar.actions`       | `ProTableAction<T>[]`                 | —                            | Configured action buttons rendered on the right side of the toolbar. Button props are passed directly on each action; use `tooltip` for tooltip content. |
| `bulkToolbar`           | `false \| { entityName?, actions? }`  | —                            | Bulk toolbar config shown when rows are selected.                                                              |
| `pagination`            | `false \| { pageSizeOptions? }`       | enabled                      | Pagination config. Set to `false` to hide pagination.                                                          |
| `dragSort`              | `boolean \| { enabled?, rowKey?, onDragSortEnd? }` | `false`       | Drag sort config. Object form keeps row key and callback grouped.                                              |
| `loading`               | `boolean`                             | `false`                      | Shows a skeleton loading state.                                                                                |
| `loadingRows`           | `number`                              | `5`                          | Number of skeleton rows to display.                                                                            |
| `emptyText`             | `ReactNode`                           | `'No data'`                  | Empty state text or custom content.                                                                            |
| `emptyIcon`             | `ReactNode`                           | inbox icon                   | Empty state icon.                                                                                              |
| `layout`                | `'full' \| 'auto'`                    | `'full'`                     | Full layout fills the available height and keeps pagination at the bottom; auto layout follows content height. |
| `stickyHeader`          | `boolean`                             | `true`                       | Keep the table header visible while the table body scrolls.                                                    |
| `columnPinning`         | `ColumnPinningState`                  | from `columns[].meta.pinned` | Controlled column pinning state. Columns can also set `meta: { pinned: 'left' \| 'right' }`.                   |
| `onColumnPinningChange` | `OnChangeFn<ColumnPinningState>`      | —                            | Called when column pinning changes.                                                                            |
| `className`             | `string`                              | —                            | Additional className for the outer container.                                                                  |

### Column Meta

| Meta                    | Type                                  | Default                      | Description                                                                                                    |
| ----------------------- | ------------------------------------- | ---------------------------- | -------------------------------------------------------------------------------------------------------------- |
| `search`                | `boolean \| object`                   | —                            | Enables the toolbar search input from this column.                                                             |
| `searchPlaceholder`     | `string`                              | `Search {columnId}...`       | Placeholder for the auto-rendered search input.                                                                |
| `filters`               | `{ label, value }[]`                  | —                            | Options used to auto-render a FacetedFilter and map values to labels in cells.                                  |
| `filterMode`            | `'single' \| 'multi'`                 | `'multi'`                    | Selection mode for the auto-rendered FacetedFilter.                                                            |
| `filterVariant`         | `'badge' \| 'text'`                   | `'badge'`                    | Cell display style when `filters` is set.                                                                      |
| `pinned`                | `'left' \| 'right'`                   | —                            | Initial pinned position for the column.                                                                        |

### useProTableUrlState

Use URL search params as initial table state and sync later table state changes back to the URL — framework-agnostic via a `search` + `navigate` interface (compatible with TanStack Router, Next.js App Router, React Router, etc.).

```tsx
import { useProTableUrlState } from '@/components/pro/pro-table/use-pro-table-url-state'

const urlState = useProTableUrlState({
  search: { page: 2, filter: 'alice' },  // current URL search params
  navigate: ({ search }) => { /* push new params to router */ },
})

<ProTable columns={columns} data={users} {...urlState} />
```

**Parameters:**

| Option          | Type            | Default      | Description                                      |
| --------------- | --------------- | ------------ | ------------------------------------------------ |
| `search`        | `SearchRecord`  | —            | Current URL search params as a plain object.     |
| `navigate`      | `NavigateFn`    | —            | Called with updated params to push to the router.|
| `pagination.pageKey`      | `string` | `'page'`    | URL key for current page.            |
| `pagination.pageSizeKey`  | `string` | `'pageSize'`| URL key for page size.               |
| `pagination.defaultPage`  | `number` | `1`         | Default page (omitted from URL).     |
| `pagination.defaultPageSize` | `number` | `10`     | Default page size (omitted from URL).|
| `columnFilters` | `Array<ColumnFilterConfig>` | `[]`  | Column filter definitions with URL key mapping. |

**Returns:**

| Key                    | Type                          | Description                               |
| ---------------------- | ----------------------------- | ----------------------------------------- |
| `defaultState`         | `Partial<ProTableState>`      | Pass to `<ProTable defaultState={...} />`. |
| `onStateChange`        | `(state) => void`             | Pass to `<ProTable onStateChange={...} />`. |
| `ensurePageInRange`    | `(pageCount: number) => void` | Call after data loads to clamp the page.  |

**Example with Next.js App Router:**

```tsx
'use client'

import { useRouter, useSearchParams } from 'next/navigation'
import { useProTableUrlState } from '@/components/pro/pro-table/use-pro-table-url-state'
import { ProTable } from '@/components/pro/pro-table'

export default function UsersTable() {
  const router = useRouter()
  const searchParams = useSearchParams()

  const search = Object.fromEntries(searchParams.entries())

  const tableState = useProTableUrlState({
    search,
    navigate: ({ search: next }) => {
      const params = new URLSearchParams()
      const resolved = typeof next === 'function' ? next(search) : next
      for (const [k, v] of Object.entries(resolved)) {
        if (v !== undefined) params.set(k, String(v))
      }
      router.push(`?${params.toString()}`)
    },
  })

  return (
    <ProTable
      {...tableState}
      columns={columns}
      data={data}
    />
  )
}
```
