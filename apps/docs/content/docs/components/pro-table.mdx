## Usage

```tsx
import { ProTable } from "@/components/pro/pro-table"

const columns = [
  {
    accessorKey: "name",
    header: "Name",
    enableSorting: true,
    meta: { pinned: "left" },
  },
  { accessorKey: "email", header: "Email" },
  { accessorKey: "createdAt", header: "Created", meta: { pinned: "right" } },
]

export default function UsersTable() {
  return (
    <ProTable
      columns={columns}
      data={users}
      searchKey="name"
      searchPlaceholder="Search by name..."
      header={<h2 className="text-lg font-semibold">Users</h2>}
      stickyHeader
      onRefresh={() => refetch()}
    />
  )
}
```

## Examples

### Pinned Columns and Sticky Header

Use `meta.pinned` on column definitions to keep important columns visible during horizontal scroll. Users can also pin or unpin columns from the column settings menu. The table header is sticky by default; set `stickyHeader={false}` to disable it.

```tsx
const columns = [
  { id: "select", size: 48, meta: { pinned: "left" } },
  { accessorKey: "name", header: "Name", size: 180, meta: { pinned: "left" } },
  { accessorKey: "email", header: "Email", size: 240 },
  { accessorKey: "createdAt", header: "Created", size: 150, meta: { pinned: "right" } },
]

<ProTable columns={columns} data={users} />
```

### Filtering with Reset

Add `meta.filters` to any column — ProTable **automatically** renders a `FacetedFilter` (Popover + Checkbox) in the toolbar, maps `value → label` in cells, and shows a **Reset** button whenever any filter is active.

- Default mode is **multi-select** — use `filterFn: "multiValueFilter"` (built-in)
- Set `meta.filterMode: "single"` for single-select (radio-like), use `filterFn: "equals"`

<ComponentPreview name="pro-table-filter-demo" previewClassName="min-h-[460px]" />

```tsx
const columns: ColumnDef<User>[] = [
  {
    accessorKey: "role",
    header: "Role",
    filterFn: "multiValueFilter", // built-in, handles string[] filter value
    meta: {
      filterPlaceholder: "Role",
      filters: [
        { label: "Admin", value: "admin" },
        { label: "Editor", value: "editor" },
      ],
    },
  },
  {
    accessorKey: "status",
    header: "Status",
    filterFn: "equals",
    meta: {
      filterPlaceholder: "Status",
      filterMode: "single",   // single-select
      filterVariant: "text",  // cell renders as plain text (default: 'badge')
      filters: [
        { label: "Active", value: "active" },
        { label: "Inactive", value: "inactive" },
      ],
    },
  },
]

// No filterRender needed — FacetedFilters are auto-rendered from meta
<ProTable columns={columns} data={data} searchKey="name" />
```

You can still pass `filterRender` (ReactNode or function) for fully custom controls.

### Bulk Actions

Render bottom-floating bulk actions when rows are selected. The floating toolbar supports keyboard navigation and `Escape` clears the selection.

```tsx
<ProTable
  columns={columns}
  data={users}
  bulkActionEntityName="user"
  bulkActionRender={({ selectedRows }) => (
    <Button size="sm" variant="outline">
      Export {selectedRows.length}
    </Button>
  )}
/>
```

### Drag-to-Sort

Enable row drag-to-sort with the `dragSort` prop. Use `rowKey` to specify the unique identifier field and `onDragSortEnd` to receive the reordered data.

Drag sorting reorders the backing `data` array by the currently rendered rows. When the table is filtered or paginated, the dragged row moves relative to the visible target row in the original data array.

<ComponentPreview
  name="pro-table-drag-sort-demo"
  previewClassName="min-h-[520px]"
/>

```tsx
<ProTable
  columns={columns}
  data={data}
  dragSort
  rowKey="id"
  onDragSortEnd={(newData) => console.log(newData)}
/>
```

## API Reference

### ProTable

| Prop                    | Type                                  | Default                      | Description                                                                                                    |
| ----------------------- | ------------------------------------- | ---------------------------- | -------------------------------------------------------------------------------------------------------------- |
| `columns`               | `ColumnDef<T>[]`                      | —                            | TanStack Table column definitions.                                                                             |
| `data`                  | `T[]`                                 | —                            | Table data array.                                                                                              |
| `header`                | `ReactNode \| (context) => ReactNode` | —                            | Custom content rendered above the toolbar.                                                                     |
| `searchKey`             | `string`                              | —                            | Column key to enable text filtering on.                                                                        |
| `searchPlaceholder`     | `string`                              | `'Search...'`                | Placeholder text for the search input.                                                                         |
| `filterRender`          | `ReactNode \| (table) => ReactNode`   | —                            | Custom filter controls rendered after the auto filters. Pass a function to access the table instance. |
| `filterMode` *(meta)*   | `'single' \| 'multi'`                 | `'multi'`                    | Column meta — selection mode for the auto-rendered FacetedFilter.                                    |
| `filterVariant` *(meta)*| `'badge' \| 'text'`                   | `'badge'`                    | Column meta — cell display style when `meta.filters` is set.                                         |
| `toolBarRender`         | `() => ReactNode[]`                   | —                            | Extra action buttons rendered on the right side of the toolbar.                                                |
| `bulkActionRender`      | `(context) => ReactNode`              | —                            | Actions shown in the bulk toolbar when rows are selected.                                                      |
| `bulkActionEntityName`  | `string`                              | `'row'`                      | Singular entity name used in the bulk toolbar selection count.                                                 |
| `showColumnToggle`      | `boolean`                             | `true`                       | Show the column visibility toggle button.                                                                      |
| `onRefresh`             | `() => void`                          | —                            | Refresh callback — shows a refresh button when provided.                                                       |
| `pageSizeOptions`       | `number[]`                            | `[10,20,50,100]`             | Page size options for the pagination control.                                                                  |
| `dragSort`              | `boolean`                             | `false`                      | Enable row drag-to-sort (disables column header click-to-sort).                                                |
| `rowKey`                | `keyof T`                             | —                            | Unique field used as the drag sort key (falls back to row index).                                              |
| `onDragSortEnd`         | `(newData: T[]) => void`              | —                            | Called after drag sort completes with the reordered data.                                                      |
| `loading`               | `boolean`                             | `false`                      | Shows a skeleton loading state.                                                                                |
| `loadingRows`           | `number`                              | `5`                          | Number of skeleton rows to display.                                                                            |
| `emptyText`             | `ReactNode`                           | `'No data'`                  | Empty state text or custom content.                                                                            |
| `emptyIcon`             | `ReactNode`                           | inbox icon                   | Empty state icon.                                                                                              |
| `showPagination`        | `boolean`                             | `true`                       | Show or hide the built-in pagination bar.                                                                      |
| `layout`                | `'full' \| 'auto'`                    | `'full'`                     | Full layout fills the available height and keeps pagination at the bottom; auto layout follows content height. |
| `stickyHeader`          | `boolean`                             | `true`                       | Keep the table header visible while the table body scrolls.                                                    |
| `columnPinning`         | `ColumnPinningState`                  | from `columns[].meta.pinned` | Controlled column pinning state. Columns can also set `meta: { pinned: 'left' \| 'right' }`.                   |
| `onColumnPinningChange` | `OnChangeFn<ColumnPinningState>`      | —                            | Called when column pinning changes.                                                                            |
| `className`             | `string`                              | —                            | Additional className for the outer container.                                                                  |
| `pagination`            | `PaginationState`                     | internal                     | Controlled pagination state (`{ pageIndex, pageSize }`). Pair with `onPaginationChange`. |
| `onPaginationChange`    | `OnChangeFn<PaginationState>`         | —                            | Called when page or page size changes.                                                   |
| `sorting`               | `SortingState`                        | internal                     | Controlled sorting state. Pair with `onSortingChange`.                                   |
| `onSortingChange`       | `OnChangeFn<SortingState>`            | —                            | Called when sort column or direction changes.                                             |
| `total`                 | `number`                              | —                            | Total row count for server-side pagination (used to compute page count).                 |
| `manualPagination`      | `boolean`                             | `false`                      | Disable client-side pagination — fetch data for the current page yourself.               |
| `manualSorting`         | `boolean`                             | `false`                      | Disable client-side sorting — sort data on the server.                                   |

### useProTableUrlState

Sync pagination, sorting, and search to URL query params via `nuqs`. Returns props you can spread directly onto `<ProTable />`.

```tsx
import { useProTableUrlState } from '@/components/ui/pro-table/use-pro-table-url-state'

const urlState = useProTableUrlState({ defaultPageSize: 20 })

// URL: ?page=2&pageSize=20&sort=name&order=asc&search=alice
```

| Option            | Type     | Default | Description                                                     |
| ----------------- | -------- | ------- | --------------------------------------------------------------- |
| `defaultPageSize` | `number` | `10`    | Initial page size.                                              |
| `prefix`          | `string` | `''`    | URL param prefix — use when multiple tables share a page.       |

The hook returns:

| Key                 | Type                          | Description                              |
| ------------------- | ----------------------------- | ---------------------------------------- |
| `pagination`        | `PaginationState`             | Pass to `<ProTable pagination={...} />`. |
| `onPaginationChange`| `OnChangeFn<PaginationState>` | Pass to `<ProTable onPaginationChange={...} />`. |
| `sorting`           | `SortingState`                | Pass to `<ProTable sorting={...} />`.    |
| `onSortingChange`   | `OnChangeFn<SortingState>`    | Pass to `<ProTable onSortingChange={...} />`. |
| `search`            | `string`                      | Current search term from the URL.        |
| `setSearch`         | `(value: string) => void`     | Update search and reset to page 1.       |

**Server-side example:**

```tsx
'use client'

import { useProTableUrlState } from '@/components/ui/pro-table/use-pro-table-url-state'
import { ProTable } from '@/components/ui/pro-table'
import useSWR from 'swr'

export default function UsersTable() {
  const urlState = useProTableUrlState()
  const { pagination, sorting, search } = urlState

  const { data, isLoading } = useSWR(
    ['/api/users', pagination, sorting, search],
    () => fetchUsers({ pagination, sorting, search }),
  )

  return (
    <ProTable
      {...urlState}
      columns={columns}
      data={data?.rows ?? []}
      total={data?.total}
      loading={isLoading}
      manualPagination
      manualSorting
    />
  )
}
```

**NuqsAdapter** is required at the app level for `nuqs` to work:

```tsx
// app/layout.tsx
import { NuqsAdapter } from 'nuqs/adapters/next/app'

export default function RootLayout({ children }) {
  return <NuqsAdapter>{children}</NuqsAdapter>
}
```
